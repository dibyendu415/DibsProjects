bool isPrime(ll n)
{   if (n <= 1)  return false;
    if (n <= 3)  return true;
    if (n % 2 == 0 or n % 3 == 0)  return false;
    ll i = 5;
    while(i * i <= n) { 
        if (n % i == 0 or n % (i + 2) == 0)  
        return false;
        i = i + 6;}
        return true;
}
ll gcd(ll a, ll b) 
{ 
    if (b == 0) 
        return a; 
    return gcd(b, a % b);  
}
